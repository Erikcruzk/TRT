        MinSum = _val;

        if (intitalized) throw;
        Log = LogFile(_log);
    }

    function Initialized() public {
        intitalized = true;
    }

    function Deposit() public payable {
        balances[msg.sender] += msg.value;
        Log.AddMessage(msg.sender, msg.value, "Put");
    }

        if (balances[msg.sender] >= MinSum && balances[msg.sender] >= _am) {
