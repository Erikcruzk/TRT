        require(tx.origin == owner);
        _;

    {
        owner = msg.sender;
        TransferLog = new Log();
    }
    
    
    
    function setLog(address _lib) onlyOwner
    {
        TransferLog = Log(_lib);
    }

    public
    payable
    {

    }

}
